보안 강화 SQL 스키마

Supabase SQL Editor에 적용하세요 (RLS 포함).

-- 평가 테이블 고도화
create table evaluations (
  id uuid default gen_random_uuid() primary key,
  employee_name text not null,
  evaluator_id uuid references auth.users(id), -- 누가 썼는지는 기록되나 RLS로 숨김
  competency int, leadership int, attitude int,
  safety int, technical int, -- 현장 항목 추가
  comment text,
  created_at timestamptz default now()
);

-- RLS: 관리자 외에는 본인이 쓴 글도 조회 불가 (완벽 익명)
alter table evaluations enable row level security;
create policy "익명 제출만 허용" on evaluations for insert with check (true);
create policy "관리자만 조회 가능" on evaluations for select 
using (auth.uid() in (select user_id from admin_users));
